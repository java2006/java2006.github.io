题解 P2590 [ZJOI2008]树的统计

题目大意：一棵大小为n()，有点权的树，共q()次操作。 每次操作询问路径和/最大值或修改点权。

建两棵线段树，一棵维护最大值，另一棵维护和，剩下就是树剖板子了。 code:

#include<bits/stdc++.h>
using namespace std;
#define N 30010
template<typename T>
inline void read(T&x){
    x=0;
    int f=1;
    char ch=getchar();
    while(ch==' '||ch=='\n')ch=getchar();
    if(ch=='-')f=-1,ch=getchar();
    while('0'<=ch&&ch<='9'){
        x*=10;
        x+=ch-'0';
        ch=getchar();
    }
    x*=f;
}
int a[N],top[N],sz[N],d[N],f[N],son[N],id[N],rk[N],tot=1,cnt=1;
struct edge{
    int v,nxt;
}e[2*N];
int head[N];
void ae(int u,int v){
    e[tot].v=v;
    e[tot].nxt=head[u];
    head[u]=tot++;
}
int n,q;
void dfs1(int u,int fa){
    sz[u]=1;
    d[u]=d[fa]+1;
    f[u]=fa;
    for(int i=head[u];i;i=e[i].nxt){
        int v=e[i].v;
        if(v==fa)continue;
        dfs1(v,u);
        sz[u]+=sz[v];
        if(sz[v]>sz[son[u]])son[u]=v;
    }
}
void dfs2(int u,int t){
    top[u]=t;
    id[u]=cnt;
    rk[cnt++]=u;
    if(!son[u])return;
    dfs2(son[u],t);
    for(int i=head[u];i;i=e[i].nxt){
        int v=e[i].v;
        if(v==f[u]||v==son[u])continue;
        dfs2(v,v);
    }
}
int mx[4*N],sm[4*N];
void build(int rt,int l,int r){
    if(l==r){
        mx[rt]=sm[rt]=a[rk[l]];
        return;
    }
    int mid=(l+r)/2;
    build(rt*2+1,l,mid);
    build(rt*2+2,mid+1,r);
    mx[rt]=max(mx[rt*2+1],mx[rt*2+2]);
    sm[rt]=sm[rt*2+1]+sm[rt*2+2];
}
void update(int rt,int l,int r,int idx,int x){
    if(l==r){
        mx[rt]=x;
        sm[rt]=x;
        return;
    }
    int mid=(l+r)/2;
    if(idx<=mid)update(rt*2+1,l,mid,idx,x);
    else update(rt*2+2,mid+1,r,idx,x);
    mx[rt]=max(mx[rt*2+1],mx[rt*2+2]);
    sm[rt]=sm[rt*2+1]+sm[rt*2+2];
}
int querymx(int rt,int l,int r,int ql,int qr){
    if(ql<=l&&r<=qr)return mx[rt];
    int mid=(l+r)/2;
    int ans=-1000000;
    if(ql<=mid)ans=max(ans,querymx(rt*2+1,l,mid,ql,qr));
    if(mid<qr)ans=max(ans,querymx(rt*2+2,mid+1,r,ql,qr));
    return ans;
}
int querysm(int rt,int l,int r,int ql,int qr){
    if(ql<=l&&r<=qr)return sm[rt];
    int mid=(l+r)/2;
    int ans=0;
    if(ql<=mid)ans+=querysm(rt*2+1,l,mid,ql,qr);
    if(mid<qr)ans+=querysm(rt*2+2,mid+1,r,ql,qr);
    return ans;
}
int main(){
    read(n);
    for(int i=0;i<n-1;i++){
        int u,v;
        read(u);read(v);
        ae(u,v);
        ae(v,u);
    }
    dfs1(1,0);
    dfs2(1,1);
    for(int i=1;i<=n;i++){
        read(a[i]);
    }
    build(0,1,n);
    read(q);
    for(int i=0;i<q;i++){
        char s[100];
        scanf("%s",s);
        int u,v;
        read(u);read(v);
        if(s[1]=='H'){//CHANGE
            update(0,1,n,id[u],v);
        }else if(s[1]=='M'){//QMAX
            int ans=-1000000;
            while(top[u]!=top[v]){
                if(d[top[u]]<d[top[v]])swap(u,v);
                ans=max(ans,querymx(0,1,n,id[top[u]],id[u]));
                u=f[top[u]];
            }
            if(d[u]<d[v])swap(u,v);
            ans=max(ans,querymx(0,1,n,id[v],id[u]));
            printf("%d\n",ans);

        }else{//QSUM
            int ans=0;
            while(top[u]!=top[v]){
                if(d[top[u]]<d[top[v]])swap(u,v);
                ans+=querysm(0,1,n,id[top[u]],id[u]);
                u=f[top[u]];
            }
            if(d[u]<d[v])swap(u,v);
            ans+=querysm(0,1,n,id[v],id[u]);
            printf("%d\n",ans);
        }
    }
    return 0;
}
